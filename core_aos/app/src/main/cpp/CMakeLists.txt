cmake_minimum_required(VERSION 3.4.1)

set(PROUDNET_ROOT ${CMAKE_SOURCE_DIR}/../../../../..)

add_library(libProudNetClient STATIC IMPORTED)
set_target_properties(libProudNetClient PROPERTIES IMPORTED_LOCATION
    ${PROUDNET_ROOT}/lib/NDK/cmake/${PROUD_CONFIG}/armeabi-v7a/libProudNetClient.a)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=gnu++11")

add_library(hello-libs SHARED
            hello-libs.cpp)

target_include_directories(hello-libs PRIVATE
                           ${PROUDNET_ROOT}/include)

target_link_libraries(hello-libs
                      android
                      libProudNetClient
                      log)
